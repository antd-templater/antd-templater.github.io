      attrs: {

// 模拟数据排序
const sortData = (data: any[], sorts: any[]) => {
  return data.sort((next: any, prev: any) => {
    for (const sort of sorts) {
      if (sort.value === 'ascend' && next[sort.field] < prev[sort.field]) {
        return -1
      }

      if (sort.value === 'ascend' && next[sort.field] > prev[sort.field]) {
        return 1
      }

      if (sort.value === 'descend' && next[sort.field] < prev[sort.field]) {
        return 1
      }

      if (sort.value === 'descend' && next[sort.field] > prev[sort.field]) {
        return -1
      }
    }

    return 1
  })
}

// 加载远程数据
