DatePicker 日期选择框

一个基于 ADatePicker 日期选择框 进行封装的单元格编辑框 => SEditCellDatePicker

无日期
2024-05-31
复制代码
显示代码
vue
<template>
  <ASpace
    :size="15"
    :wrap="true"
    style="margin: 0"
    direction="vertical"
  >
    <SEditCellDatePicker
      empty="无日期"
      :text="value1"
      :allowClear="true"
      :cellStyle="cellStyle"
      format="YYYY-MM-DD"
      valueFormat="YYYY-MM-DD"
      @confirm="(proxy) => value1 = proxy.value"
    />

    <SEditCellDatePicker
      empty="无日期"
      :text="value2"
      :allowClear="true"
      :cellStyle="cellStyle"
      format="YYYY-MM-DD"
      valueFormat="YYYY-MM-DD"
      @confirm="(proxy) => value2 = proxy.value"
    />

    <SEditCellDatePicker
      empty="无日期"
      :text="value3"
      :opened="true"
      :allowClear="true"
      :cellStyle="cellStyle"
      format="YYYY-MM-DD"
      valueFormat="YYYY-MM-DD"
      @confirm="(proxy) => value3= proxy.value"
    />
  </ASpace>
</template>

<script setup lang="ts">
const value1 = ref('')
const value2 = ref('2024-05-31')
const value3 = ref('2024-06-01')

const cellStyle = {
  inputWrapper: {
    display: 'inline-block',
    paddingRight: '30px',
    width: '180px',
  },
  textWrapper: {
    display: 'inline-block',
    paddingRight: '32px',
    width: 'auto',
  },
}
</script>
